#include<stdio.h>
#include<stdbool.h>
int main()
{
   int n,i,j,temp;
   bool flag = false;
   printf("Enter number of elements:");
   scanf("%d",&n);
   int a[n];
   printf("Enter the elments\n");
   for(i=0;i<n;i++)
   {
      scanf("%d",&a[i]);
   }
   for(i=0;i<(n-1);i++)
   {
      for(j=0;j<(n-i-1);j++)
      {
         if(a[j] > a[j+1])
         {
            temp=a[j];
            a[j]=a[j+1];
            a[j+1]=temp;
            flag=true;
         }

      }
      if(!flag)
      {
         break;
      }

   }
   printf("Sorted Array \n");
   for(i=0;i<n;i++)
   {
      printf("%d",a[i]);
      printf("\t");
   }
   return 0;
}
