#include <stdio.h>
#include <string.h>
void ShiftTable(char P[], int m, int Table[])
{
    int i, j;
    for(i = 0; i < 256; i++)
    {
        Table[i] = m;
    }
    for (j = 0; j <= m - 2; j++)
    {
        Table[(unsigned char)P[j]] = m - 1 - j;
    }
}
int HorspoolMatching(char P[], char T[])
{
    int m = strlen(P);
    int n = strlen(T);
    int Table[256];
    int i, k;
    if (m == 0) return -1;
    ShiftTable(P, m, Table);
    i = m - 1;
    while (i <= n - 1)
    {
        k = 0;
        while (k <= m - 1 && P[m - 1 - k] == T[i - k])
        {
            k = k + 1;
        }
        if (k == m)
        {
            return i - m + 1;
        }
        else
        {
            i = i + Table[(unsigned char)T[i]];
        }
    }
    return -1;
}
int main()
{
    char T[100], P[50];
    printf("Enter Text: ");
    fgets(T, sizeof(T), stdin);
    T[strcspn(T, "\n")] = '\0';
    printf("Enter Pattern: ");
    fgets(P, sizeof(P), stdin);
    P[strcspn(P, "\n")] = '\0';
    int res = HorspoolMatching(P, T);
    if (res != -1)
        printf("\nFound at index: %d\n", res);
    else
        printf("\nNot found.\n");
    return 0;
}
